def make_testsquidarray(Nsquids, w_jl, w_jr):
                   
    l1 = SD.mybox((0,0), w_l1, h_l1)
    
    l1 = translate(l1, yoff=-h_l1)
    # anker points to attach the SQUID loop
    xdummy, y_l1_anker = SD.anker(l1,
                                  'lower_left')
    x_l1_anker, ydummy = SD.anker(l1,
                                  'center')
    
    l2 = SD.mybox((0,0), w_l1, h_l1)
    
    l2 = translate(l2, yoff=-h_l1)
    # anker points to attach the SQUID loop
    xdummy, y_l2_anker = SD.anker(l2,
                                  'upper_left')
    x_l2_anker, ydummy = SD.anker(l2,
                                  'center')
    
    ############ SQUID LOOP ############
    ####################################
    
    ######### OUTER LOOP ############
    
    lo = SD.mybox((0,0), w_lo, h_lo)
    xdummy, y_lo_anker = SD.anker(lo,
                                  'upper_left')
    x_lo_anker, ydummy = SD.anker(lo,
                                  'center')
    
    #plot(x_lo_anker,y_lo_anker,'o')
    #SD.showPolygons([lo])

    ######### INNER LOOP ############
    
    loi   = SD.mybox((0,0), w_lo-w_jl-w_jr, ih_lo)
    y_loi = (h_lo-ih_lo)/2.
    x_loi = w_jl
    loi   = translate(loi, xoff=x_loi,
                            yoff = y_loi)
    
    #SD.showPolygons([loi])
    
    ######### BRIDGE ############
    
    lob   = SD.mybox((0,0), w_lo, w_b)
    y_lob = h_lo/2. - w_b/2.
    lob   = translate(lob, yoff=y_lob)
    #SD.showPolygons([lob])

    ######## UNDERCUT BOXES ###########
    ub    = SD.mybox((0,0), w_lo-w_jl-w_jr, w_u)
    x_ub  = w_jl
    y_ub1 = (h_lo - ih_lo)/2.

    y_ub2 = y_ub1 + ih_lo - w_u
    
    ub1 = translate(ub, xoff = x_ub,
                        yoff = y_ub1)
    
    ub2 = translate(ub, xoff = x_ub,
                        yoff = y_ub2)
    
    
    ######################################
    ######### FINALIZE & PLOT ############
    ######################################
    
    squid_loop = lo.difference(loi)
    squid_loop = squid_loop.difference(lob)
    squid_loop = SD.flattenMultipoly(squid_loop)
    #SD.showPolygons(squid_loop)
    squid_loop = list(squid_loop) #+ list([ub1,ub2])
    
    SQUID_loop = SD.BRAND_NEW_STRUCTURE(SD.empty(),[])
    SQUID_loop.EBL.add_polis(squid_loop)
    #SQUID_loop.add_EBL(squid_loop)
    SQUID_loop.add_anker([x_lo_anker, y_lo_anker],'upper center')
    SQUID_loop.add_anker([x_lo_anker, 0],'lower center')
    
    SQUID_loop.add_layer([ub1, ub2], 'UNDERCUT')
    
    xl, yl = SQUID_loop.ANKERS['upper center']
    x1, y1 = x_l1_anker, y_l1_anker
    
    SQUID_loop.translate([x1,y1],
                         [xl,yl])

    dummy, yl1u = SD.anker(l1, 'upper_left')
    xl1u, dummy = SD.anker(l1, 'center')
    
    lbar = SD.mybox((0,0),wbar,hbar)
    lbarx, dummy = SD.anker(lbar,'center')
    dummy, lbary = SD.anker(lbar,'lower_left')

    TESTSQUIDS = SD.BRAND_NEW_STRUCTURE([],[])
    TESTSQUIDS.add_layer([],'UNDERCUT')
    
    figure(figsize=(10,10))

    lbar1 = translate(lbar, xoff=xl1u-lbarx, yoff=yl1u-lbary)
    
    TESTSQUIDS.add_anker([xl1u,yl1u],'mid')

    TESTSQUIDS.EBL.add_polis([lbar1,l1])
    
    TESTSQUIDS.EBL.add_polis(SQUID_loop.get_polygons('EBL'))
    TESTSQUIDS.UNDERCUT.add_polis(SQUID_loop.get_polygons('UNDERCUT'))
    
    nsquids = Nsquids-1
    
    ox, oy = SQUID_loop.ANKERS['lower center']
    
    for i in arange(nsquids):
        SL = SQUID_loop.make_copy()
        nx, ny = SL.ANKERS['upper center']
        SL.translate([ox,oy], [nx,ny])
        ox, oy = SL.ANKERS['lower center']
        
        TESTSQUIDS.EBL.add_polis(SL.get_polygons('EBL'))
        TESTSQUIDS.UNDERCUT.add_polis(SL.get_polygons('UNDERCUT'))
        
        
    lend = translate(l2, xoff=ox-x_l2_anker, yoff=oy-y_l2_anker)
    lbar2 = translate(lbar, xoff=xl1u-lbarx, yoff=oy-y_l2_anker-h_l1-hbar)
    TESTSQUIDS.EBL.add_polis([lend, lbar2])
    TESTSQUIDS.add_anker([xl1u, hbar/2.], 'upper center')
    
    return TESTSQUIDS
